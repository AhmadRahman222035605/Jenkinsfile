pipeline {
    agent any

    stages {
        stage('Build') {
            steps {
                sh 'mvn clean package'
            }
        }
        
        stage('Unit and Integration Tests') {
            steps {
                sh 'mvn test'
            }
        }
        
        stage('Code Analysis') {
            steps {
                sh 'mvn checkstyle:checkstyle'
            }
        }
        
        stage('Security Scan') {
            steps {
                sh 'mvn dependency-check:check'
            }
        }
        
        stage('Deploy to Staging') {
            steps {
                // Perform the deployment to your staging server here
                // Modify the command as per your deployment requirements
                sh 'scp target/myapp.war user@staging-server:/path/to/deploy'
            }
        }
        
        stage('Integration Tests on Staging') {
            steps {
                // Run integration tests on the staging environment here
                // Modify the command as per your application's requirements
                sh 'mvn integration-test'
            }
        }
        
        stage('Deploy to Production') {
            steps {
                // Perform the deployment to your production server here
                // Modify the command as per your deployment requirements
                sh 'scp target/myapp.war user@production-server:/path/to/deploy'
            }
        }
    }
    

    post {
        always {
            // Send email notifications at the end of test and security scan stages
            emailext (
                subject: "Pipeline Status: ${currentBuild.result}",
                body: "Pipeline completed with status: ${currentBuild.result}",
                to: "ahmad.rahman003@gmail.com",
                attachmentsPattern: '**/*'
            )
        }
    }
}
